  **Add item flow (CAT-01, D-09, D-16):**
  ```typescript
  async function handleSave(data: { name: string; unit: string; unitPriceCents: number }) {
    const contractorId = useAuthStore.getState().contractor?.contractorId;
    const collection = database.get<CatalogItem>('catalog_items');
    const newItem = await database.write(async () => {
      return collection.create((record) => {
        record.contractorId = contractorId;
        record.name = data.name;
        record.unit = data.unit;
        record.unitPriceCents = data.unitPriceCents;
        record.tradeCategory = useAuthStore.getState().contractor?.trade ?? null;
        record.isArchived = false;
      });
    });
    await enqueue({
      entityType: 'catalog_item',
      entityId: newItem.id,
      action: 'create',
      payload: { name: data.name, unit: data.unit, unitPriceCents: data.unitPriceCents },
    });
    setSheetVisible(false);
  }
  ```
